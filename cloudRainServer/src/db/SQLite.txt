CREATE TABLE IF NOT EXISTS user(
id INTEGER PRIMARY KEY AUTOINCREMENT,
name VARCHAR(100) NOT NULL,
ip VARCHAR(32) NOT NULL,
date_GMT DATE NOT NULL,
date_UTC DATE NOT NULL,
status VARCHAR(100) NOT NULL
);
CREATE TABLE IF NOT EXISTS apache(
id INTEGER PRIMARY KEY AUTOINCREMENT,
id_user INTEGER,
date_GMT DATE NOT NULL,
date_UTC DATE NOT NULL,
FOREIGN KEY(id_user) REFERENCES user(id)
);
CREATE TABLE IF NOT EXISTS mysql(
id INTEGER PRIMARY KEY AUTOINCREMENT,
id_user INTEGER,
date_GMT DATE NOT NULL,
date_UTC DATE NOT NULL,
FOREIGN KEY(id_user) REFERENCES user(id)
);
CREATE TABLE IF NOT EXISTS mysql_meta(
id INTEGER PRIMARY KEY AUTOINCREMENT,
meta_key VARCHAR(100) NOT NULL,
meta_value TEXT,
id_mysql INTEGER,
FOREIGN KEY(id_mysql) REFERENCES mysql(id)
);
CREATE TABLE IF NOT EXISTS apache_meta(
id INTEGER PRIMARY KEY AUTOINCREMENT,
meta_key VARCHAR(100) NOT NULL,
meta_value TEXT,
id_apache INTEGER,
FOREIGN KEY(id_apache) REFERENCES apache(id)
);
